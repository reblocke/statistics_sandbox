Medicaid Provider Spending EDA (DuckDB + Python)

Purpose

This notebook performs SQL-first exploratory data analysis (EDA) on medicaid-provider-spending.parquet using DuckDB. It is designed for a large file and keeps memory use low by:

  1. Reading parquet lazily through DuckDB.
  2. Aggregating in SQL first.
  3. Moving only small summary tables to pandas for plotting.

def qident(name: str) -> str:
    return '"' + name.replace('"', '""') + '"'


def is_numeric_duckdb_type(type_name: str) -> bool:
    t = str(type_name).upper()
    numeric_markers = (
        "TINYINT",
        "SMALLINT",
        "INTEGER",
        "BIGINT",
        "HUGEINT",
        "UTINYINT",
        "USMALLINT",
        "UINTEGER",
        "UBIGINT",
        "UHUGEINT",
        "FLOAT",
        "DOUBLE",
        "REAL",
        "DECIMAL",
    )
    return any(marker in t for marker in numeric_markers)


def detect_column_roles(schema_df: pd.DataFrame) -> dict[str, list[str]]:
    """
    Detect semantic roles from column names and SQL types.
    Returns role -> list of candidate column names.
    """
    patterns = {
        "provider_id": re.compile(r"(?:^|_)(npi|provider(_id)?|prov(_id)?|prescriber|billing)(?:$|_)", re.I),
        "year": re.compile(r"(?:^|_)(year|yr|fiscal_year|calendar_year)(?:$|_)", re.I),
        "date": re.compile(r"(?:^|_)(date|month|quarter|period|service_date)(?:$|_)", re.I),
        "geo": re.compile(r"(?:^|_)(state|county|zip|zipcode|fips|region|city)(?:$|_)", re.I),
        "spend": re.compile(r"(?:^|_)(spend|payment|paid|allowed|reimb|charge|cost|amount|total)(?:$|_)", re.I),
        "volume": re.compile(r"(?:^|_)(claim|service|visit|encounter|beneficiary|bene|member|patient|count|volume|utilization)(?:$|_)", re.I),
        "specialty": re.compile(r"(?:^|_)(specialty|provider_type|prvdr_type)(?:$|_)", re.I),
        "taxonomy": re.compile(r"(?:^|_)(taxonomy)(?:$|_)", re.I),
        "id": re.compile(r"(?:^|_)(id|identifier|key)(?:$|_)", re.I),
    }

    roles: dict[str, list[str]] = {key: [] for key in patterns}
    roles["numeric"] = []
    roles["categorical"] = []

    for row in schema_df.itertuples(index=False):
        col = str(row.column_name)
        col_type = str(row.column_type)
        col_lower = col.lower()

        if is_numeric_duckdb_type(col_type):
            roles["numeric"].append(col)
        else:
            roles["categorical"].append(col)

        for role, pattern in patterns.items():
            if pattern.search(col_lower):
                roles[role].append(col)

    # De-duplicate while preserving order
    for k, v in roles.items():
        roles[k] = list(dict.fromkeys(v))

    return roles


def first_existing(candidates: list[str], all_columns: list[str]) -> Optional[str]:
    for c in candidates:
        if c in all_columns:
            return c
    return None


def safe_ratio(numerator: float, denominator: float) -> float:
    if denominator is None or pd.isna(denominator) or denominator == 0:
        return np.nan
    return float(numerator) / float(denominator)


def build_provider_long_view(scope: str, source_relation: str = "medicaid_spending") -> str:
    scope = scope.lower()
    if scope not in {"either", "billing", "servicing"}:
        raise ValueError(f"Unsupported scope: {scope}")

    billing_select = f"""
        SELECT
            CAST({qident(BILLING_NPI_COL)} AS VARCHAR) AS provider_npi,
            CAST({qident(MONTH_COL)} AS VARCHAR) AS claim_month,
            CAST({qident(HCPCS_COL)} AS VARCHAR) AS hcpcs_code,
            TRY_CAST({qident(CLAIMS_COL)} AS DOUBLE) AS total_claims,
            TRY_CAST({qident(PAID_COL)} AS DOUBLE) AS total_paid,
            'billing' AS provider_role
        FROM {source_relation}
        WHERE {qident(BILLING_NPI_COL)} IS NOT NULL
    """

    servicing_select = f"""
        SELECT
            CAST({qident(SERVICING_NPI_COL)} AS VARCHAR) AS provider_npi,
            CAST({qident(MONTH_COL)} AS VARCHAR) AS claim_month,
            CAST({qident(HCPCS_COL)} AS VARCHAR) AS hcpcs_code,
            TRY_CAST({qident(CLAIMS_COL)} AS DOUBLE) AS total_claims,
            TRY_CAST({qident(PAID_COL)} AS DOUBLE) AS total_paid,
            'servicing' AS provider_role
        FROM {source_relation}
        WHERE {qident(SERVICING_NPI_COL)} IS NOT NULL
    """

    if scope == "billing":
        return billing_select
    if scope == "servicing":
        return servicing_select

    return f"""
        {billing_select}
        UNION ALL
        {servicing_select}
          AND ({qident(BILLING_NPI_COL)} IS NULL OR {qident(SERVICING_NPI_COL)} <> {qident(BILLING_NPI_COL)})
    """
